# For-loops may have different steps

* Reverse for loop: 

``` js live
for (let i = 5; i >= 0; i--){
 console.log(i);
}
```

* For loop with a step of 2

``` js live
for (let j = 1; j <= 10; j += 2) {
   console.log(j);
}
```

* For loop which multiplies itself by 2 with each step

``` js live
for (let k = 1; k <= 6; k *= 2) {
   console.log(k);
}
```

* For loop which uses a step which divides the variable by 2

``` js live
for (let d = 40; d > 0; d /= 2) {

}
```

```
function numbersFromNto1(input) {
  let n = Number(input);
  let result = '';
  for (let i = n; i >= 1; i--) {
    if (i < n)
      result += ", ";
    result += i;
  }
  console.log(result);
}
```

```
function evenPowersOfTwo(n) {
  let num = 1;
  let result = '';
  for (let i = 0; i <= n; i += 2) {
    if (i > 0)
      result += ", ";
    result += num;
    num = num * 2 * 2;
  }
  console.log(result);
}
```
